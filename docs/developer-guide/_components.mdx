export function Callout({ title, tone = "info", children }) {
  const toneMap = {
    info: "border-sky-500/40 bg-sky-500/10",
    success: "border-emerald-500/40 bg-emerald-500/10",
    warning: "border-amber-500/40 bg-amber-500/10",
    danger: "border-rose-500/40 bg-rose-500/10",
  };

  return (
    <div className={`my-6 rounded-xl border px-4 py-3 ${toneMap[tone] || toneMap.info}`}>
      <p className="mb-1 text-sm font-semibold">{title}</p>
      <div className="text-sm leading-6">{children}</div>
    </div>
  );
}

export function CommandPanel({ title, commands }) {
  return (
    <div className="my-6 overflow-hidden rounded-xl border border-slate-700">
      <div className="border-b border-slate-700 bg-slate-900 px-4 py-2 text-xs font-semibold uppercase tracking-wide text-slate-300">
        {title}
      </div>
      <pre className="overflow-x-auto bg-slate-950 px-4 py-3 text-sm text-slate-100">
        <code>{commands.join("\n")}</code>
      </pre>
    </div>
  );
}
